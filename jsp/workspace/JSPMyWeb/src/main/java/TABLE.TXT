CREATE USER JSP IDENTIFIED BY JSP;
GRANT RESOURCE, CONNECT TO JSP;
ALTER USER JSP DEFAULT TABLESPACE USERS QUOTA UNLIMITED ON USERS;

CREATE TABLE USERS (
    ID VARCHAR2(30),
    PW VARCHAR2(30),
    NAME VARCHAR2(30),
    EMAIL VARCHAR2(50),
    ADDRESS VARCHAR2(200),
    GENDER CHAR(1) DEFAULT 'M',
    REGDATE DATE DEFAULT SYSDATE
);

ALTER TABLE USERS ADD CONSTRAINT USERS_PK PRIMARY KEY (ID);
-- ALTER TABLE USERS ADD CONSTRAINT USERS_UK UNIQUE (EMAIL);



CREATE TABLE BOARD (
    BNO NUMBER(10),      --PK
    WRITER VARCHAR2(30) NOT NULL, --작성자
    TITLE VARCHAR2(50) NOT NULL,  --제목
    CONTENT VARCHAR2(500), --내용
    HIT NUMBER(10),
    REGDATE DATE DEFAULT SYSDATE
);
ALTER TABLE BOARD ADD CONSTRAINT BOARD_PK PRIMARY KEY (BNO);
CREATE SEQUENCE BOARD_SEQ NOCACHE;
